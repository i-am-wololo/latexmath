\section{Suites}
	\begin{property}{Suites arithmetique}
		terme $u_n = an+u_0$, o\`u a est la raison \\
		somme $n\frac{u_0+u_n}{2}$
	\end{property}
	\begin{property}{Suites geo}
		terme  $u_n=u_0*q^n$, o\`u q est la raison \\
		somme  $\frac{1-q^{n+1}}{1-q}$ 
	\end{property}
